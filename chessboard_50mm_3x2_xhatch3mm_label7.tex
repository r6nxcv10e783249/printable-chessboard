% FIDE Tournament Chessboard
% Squares    : 50 mm (5.0 cm)
% Layout     : 3x2 A4 mosaic (P1 P2 P3 / P4 P5 P6)
% Dark fill  : crosshatch 45°+135°, line spacing 3.0 mm, line weight 0.30 pt
% Labels     : 16pt bold, external border strip (1.2 cm), style 7 — black!80 on white opaque badge, no border
% Margin     : 1 cm on all outer edges
%
% Board geometry:
%   sq=5.0 cm, border=1.2 cm, total=42.4 cm x 42.4 cm
%   tile        = 20.0 x 28.7 cm
%   col_step    = 11.7 cm  =>  col_offsets = [0, 11.7, 23.4]
%   row_step    = 14.7 cm  =>  row_offsets = [0, 14.7]
%   overlap_h   =  8.3 cm
%   overlap_v   = 14.0 cm
%
%   Outer frame exactly 1 cm from page edge on all outer sides:
%     P1: left+top   P2: top        P3: right+top
%     P4: left+bot   P5: bottom     P6: right+bot
%
%   Content per page:
%     P1,P4: files a-d   P2,P5: files c-f   P3,P6: files e-h
%     P1,P2,P3: ranks 8-4   P4,P5,P6: ranks 5-1

\documentclass[a4paper]{article}

\usepackage[
  a4paper,
  top=0mm, bottom=0mm, left=0mm, right=0mm,
  noheadfoot, nomarginpar
]{geometry}

\usepackage{tikz}
\usepackage{helvet}
\renewcommand{\familydefault}{\sfdefault}

\pagestyle{empty}

\def\sq{5.0}
\def\bd{1.2}
\def\btotal{42.4}

% -----------------------------------------------------------------------
% Crosshatch macro — local coords (0,0)-(sq,sq)
% 45°+135°, spacing 3.0 mm, step along axis = 3.0*sqrt(2) = 0.4243 cm
% Line list covers -sq to +2*sq to ensure full square coverage after clip
% -----------------------------------------------------------------------
\newcommand{\crosshatch}{%
  \foreach \cc in {
    -5.00,-4.58,-4.15,-3.72,-3.30,-2.87,-2.44,-2.02,-1.59,
    -1.16,-0.74,-0.31, 0.12, 0.55, 0.97, 1.40, 1.83, 2.25,
     2.68, 3.11, 3.53, 3.96, 4.39, 4.81, 5.24}{%
    \draw[black, line width=0.30pt](0,\cc)--(\sq,\cc+\sq);%
    \draw[black, line width=0.30pt](0,\cc+\sq)--(\sq,\cc);}%
}

% -----------------------------------------------------------------------
% \drawpage{xoffset}{yoffset}
%   Full A4 tikzpicture (21 x 29.7 cm, y grows downward).
%   Board origin at page coords (1.0 - xoff, 1.0 - yoff).
%   Dark squares: white base fill + clipped crosshatch in shifted scope.
%   Light squares: white fill only.
% -----------------------------------------------------------------------
\newcommand{\drawpage}[2]{%
  \pgfmathsetmacro{\ssx}{1.0 - #1}%
  \pgfmathsetmacro{\ssy}{1.0 - #2}%
  \noindent%
  \begin{tikzpicture}[x=1cm, y=-1cm]

    \useasboundingbox (0,0) rectangle (21, 29.7);
    \clip (0,0) rectangle (21, 29.7);
    \fill[white] (0,0) rectangle (21,29.7);

    % ---- 64 squares ----
    % a1 = file i=0, rank_row j=7: (0+7)=7 odd => dark ✓
    \foreach \i in {0,...,7}{
      \foreach \j in {0,...,7}{
        \pgfmathsetmacro{\dk}{int(mod(\i+\j,2))}%
        \ifnum\dk=0
          % Light square — white
          \fill[white]
            ({\bd+\i*\sq+\ssx}, {\bd+\j*\sq+\ssy})
            rectangle
            ({\bd+\i*\sq+\sq+\ssx}, {\bd+\j*\sq+\sq+\ssy});
        \else
          % Dark square — white base then crosshatch in local shifted scope
          \fill[white]
            ({\bd+\i*\sq+\ssx}, {\bd+\j*\sq+\ssy})
            rectangle
            ({\bd+\i*\sq+\sq+\ssx}, {\bd+\j*\sq+\sq+\ssy});
          \begin{scope}[shift={({\bd+\i*\sq+\ssx},{\bd+\j*\sq+\ssy})}]
            \begin{scope}
              \clip (0,0) rectangle (\sq,\sq);
              \crosshatch
            \end{scope}
          \end{scope}
        \fi
      }
    }

    % ---- Internal grid lines ----
    \foreach \k in {1,...,7}{
      \draw[black, line width=0.3pt]
        ({\bd+\k*\sq+\ssx}, {\bd+\ssy}) -- ({\bd+\k*\sq+\ssx}, {\bd+8*\sq+\ssy});
      \draw[black, line width=0.3pt]
        ({\bd+\ssx}, {\bd+\k*\sq+\ssy}) -- ({\bd+8*\sq+\ssx}, {\bd+\k*\sq+\ssy});
    }

    % ---- Playing area border ----
    \draw[black, line width=1.2pt]
      ({\bd+\ssx},{\bd+\ssy}) rectangle ({\bd+8*\sq+\ssx},{\bd+8*\sq+\ssy});

    % ---- Outer frame (enclosing board + label strip) ----
    \draw[black, line width=0.8pt]
      ({\ssx},{\ssy}) rectangle ({\btotal+\ssx},{\btotal+\ssy});

    % ---- Coordinate labels (16pt bold, in border strip) ----
    {\fontsize{16}{18}\selectfont\bfseries

    % Files — bottom border (normal)
    \foreach \i/\ltr in {0/a,1/b,2/c,3/d,4/e,5/f,6/g,7/h}{
      \node[anchor=center, fill=white, inner sep=1.2pt, text=black!80] at
        ({\bd+\i*\sq+\sq/2+\ssx}, {\bd+8*\sq+0.6+\ssy}) {\ltr};
    }

    % Files — top border (rotated 180°)
    \foreach \i/\ltr in {0/a,1/b,2/c,3/d,4/e,5/f,6/g,7/h}{
      \node[anchor=center, rotate=180, fill=white, inner sep=1.2pt, text=black!80] at
        ({\bd+\i*\sq+\sq/2+\ssx}, {0.6+\ssy}) {\ltr};
    }

    % Ranks — left border (normal): j=0=>rank8, j=7=>rank1
    \foreach \j/\num in {0/8,1/7,2/6,3/5,4/4,5/3,6/2,7/1}{
      \node[anchor=center, fill=white, inner sep=1.2pt, text=black!80] at
        ({0.6+\ssx}, {\bd+\j*\sq+\sq/2+\ssy}) {\num};
    }

    % Ranks — right border (rotated 180°)
    \foreach \j/\num in {0/8,1/7,2/6,3/5,4/4,5/3,6/2,7/1}{
      \node[anchor=center, rotate=180, fill=white, inner sep=1.2pt, text=black!80] at
        ({\bd+8*\sq+0.6+\ssx}, {\bd+\j*\sq+\sq/2+\ssy}) {\num};
    }

    }% end font group

  \end{tikzpicture}%
}

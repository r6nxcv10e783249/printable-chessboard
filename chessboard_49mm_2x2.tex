% FIDE-style Chessboard — 49mm squares, labels inside edge squares, 2x2 A4 mosaic
%
% Design:
%   - No outer border strip: board = 8*4.9cm = 39.2 x 39.2 cm
%   - Labels inside outermost squares (file letters + rank numbers)
%   - White labels on dark squares; 30% grey on light squares
%   - Labels rotated 180 on rank-8 (top) and file-h (right) for Black
%   - Font: 9pt bold Helvetica
%   - 5mm printer-safe margin on all outer edges
%
% Geometry:
%   sq=4.9cm  board=39.2cm  margin=0.5cm
%   tile        = 20.5 x 29.2 cm
%   col_step    = 19.2 cm  =>  col_offsets = [0, 19.2]
%   row_step    = 10.5 cm  =>  row_offsets = [0, 10.5]
%   overlap_h   =  1.3 cm  (26.5% of a square — easily visible for alignment)
%   overlap_v   = 18.7 cm
%
%   All four outer page edges: board frame exactly 5mm from page edge.
%   Compatible with standard home/office printers (non-printable zone <= 5mm).
%
% Content per page:
%   P1: files a-e, ranks 8-3   P2: files d-h, ranks 8-3
%   P3: files a-e, ranks 6-1   P4: files d-h, ranks 6-1
%
% Assembly (P1 P2 / P3 P4):
%   Align P1+P2 by matching the shared squares in files d-e.
%   Align top row with bottom row by matching ranks 6-3 (shared zone).

\documentclass[a4paper]{article}

\usepackage[
  a4paper,
  top=0mm, bottom=0mm, left=0mm, right=0mm,
  noheadfoot, nomarginpar
]{geometry}

\usepackage{tikz}
\usepackage{helvet}
\renewcommand{\familydefault}{\sfdefault}

\pagestyle{empty}

\def\sq{4.9}     % square size cm
\def\btotal{39.2} % total board cm (8 * sq)
\def\lpad{0.22}  % label padding from square edge cm

% -----------------------------------------------------------------------
% \drawpage{xoffset}{yoffset}
%   Board origin at page coords (0.5 - xoff, 0.5 - yoff).
%   Full A4 tikzpicture, clipped to page boundary, y grows downward.
% -----------------------------------------------------------------------
\newcommand{\drawpage}[2]{%
  \pgfmathsetmacro{\ssx}{0.5 - #1}%
  \pgfmathsetmacro{\ssy}{0.5 - #2}%
  \noindent%
  \begin{tikzpicture}[x=1cm, y=-1cm]

    \useasboundingbox (0,0) rectangle (21,29.7);
    \clip (0,0) rectangle (21,29.7);
    \fill[white] (0,0) rectangle (21,29.7);

    % ---- 64 squares ----
    % a1 = i=0,j=7: (0+7)=7 odd => dark ✓
    \foreach \i in {0,...,7}{
      \foreach \j in {0,...,7}{
        \pgfmathsetmacro{\dk}{int(mod(\i+\j,2))}
        \ifnum\dk=1
          \fill[black!35]
            ({\i*\sq+\ssx},{\j*\sq+\ssy})
            rectangle
            ({\i*\sq+\sq+\ssx},{\j*\sq+\sq+\ssy});
        \fi
      }
    }

    % ---- Grid lines (all 9 verticals and 9 horizontals) ----
    \foreach \k in {0,...,8}{
      \draw[black, line width=0.3pt]
        ({\k*\sq+\ssx},{\ssy}) -- ({\k*\sq+\ssx},{8*\sq+\ssy});
      \draw[black, line width=0.3pt]
        ({\ssx},{\k*\sq+\ssy}) -- ({8*\sq+\ssx},{\k*\sq+\ssy});
    }

    % ---- Outer playing area border ----
    \draw[black, line width=1.2pt]
      ({\ssx},{\ssy}) rectangle ({8*\sq+\ssx},{8*\sq+\ssy});

    % ---- Labels inside edge squares ----
    % File letters: rank 1 (j=7) bottom row and rank 8 (j=0) top row
    % Rank numbers: file a (i=0) left col  and file h (i=7) right col
    % anchor=south places text above the anchor point;
    % with y=-1cm axis, "south" is toward increasing y (downward in board).
    % We want the label near the BOTTOM of the square for rank-1 row:
    %   y-position = j*sq + sq - lpad  (near bottom edge of square)
    % and near the TOP for rank-8 row (rotated):
    %   y-position = j*sq + lpad       (near top edge of square)

    % --- File labels: rank 1 row (j=7), normal orientation ---
    \foreach \i/\ltr in {0/a,1/b,2/c,3/d,4/e,5/f,6/g,7/h}{
      \pgfmathsetmacro{\dk}{int(mod(\i+7,2))}
      \ifnum\dk=1
        \fill[white, opacity=0] (0,0) rectangle (0,0); % dummy to reset color
        {\fontsize{9}{10}\selectfont\bfseries\color{white}
         \node[anchor=south] at ({\i*\sq+\sq/2+\ssx},{7*\sq+\sq-\lpad+\ssy}) {\ltr};}
      \else
        {\fontsize{9}{10}\selectfont\bfseries\color{black!70}
         \node[anchor=south] at ({\i*\sq+\sq/2+\ssx},{7*\sq+\sq-\lpad+\ssy}) {\ltr};}
      \fi
    }

    % --- File labels: rank 8 row (j=0), rotated 180 for Black ---
    \foreach \i/\ltr in {0/a,1/b,2/c,3/d,4/e,5/f,6/g,7/h}{
      \pgfmathsetmacro{\dk}{int(mod(\i+0,2))}
      \ifnum\dk=1
        {\fontsize{9}{10}\selectfont\bfseries\color{white}
         \node[anchor=south, rotate=180] at ({\i*\sq+\sq/2+\ssx},{0*\sq+\lpad+\ssy}) {\ltr};}
      \else
        {\fontsize{9}{10}\selectfont\bfseries\color{black!70}
         \node[anchor=south, rotate=180] at ({\i*\sq+\sq/2+\ssx},{0*\sq+\lpad+\ssy}) {\ltr};}
      \fi
    }

    % --- Rank labels: file a col (i=0), normal orientation ---
    \foreach \j/\num in {0/8,1/7,2/6,3/5,4/4,5/3,6/2,7/1}{
      \pgfmathsetmacro{\dk}{int(mod(0+\j,2))}
      \ifnum\dk=1
        {\fontsize{9}{10}\selectfont\bfseries\color{white}
         \node[anchor=west] at ({0*\sq+\lpad+\ssx},{\j*\sq+\sq/2+\ssy}) {\num};}
      \else
        {\fontsize{9}{10}\selectfont\bfseries\color{black!70}
         \node[anchor=west] at ({0*\sq+\lpad+\ssx},{\j*\sq+\sq/2+\ssy}) {\num};}
      \fi
    }

    % --- Rank labels: file h col (i=7), rotated 180 for Black ---
    \foreach \j/\num in {0/8,1/7,2/6,3/5,4/4,5/3,6/2,7/1}{
      \pgfmathsetmacro{\dk}{int(mod(7+\j,2))}
      \ifnum\dk=1
        {\fontsize{9}{10}\selectfont\bfseries\color{white}
         \node[anchor=west, rotate=180] at ({7*\sq+\sq-\lpad+\ssx},{\j*\sq+\sq/2+\ssy}) {\num};}
      \else
        {\fontsize{9}{10}\selectfont\bfseries\color{black!70}
         \node[anchor=west, rotate=180] at ({7*\sq+\sq-\lpad+\ssx},{\j*\sq+\sq/2+\ssy}) {\num};}
      \fi
    }

  \end{tikzpicture}%
}

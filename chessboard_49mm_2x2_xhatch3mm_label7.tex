% FIDE-style Chessboard
% Squares    : 49 mm (4.9 cm)
% Layout     : 2x2 A4 mosaic (P1 P2 / P3 P4)
% Dark fill  : crosshatch 45°+135°, line spacing 3.0 mm, line weight 0.30 pt
% Labels     : Style 7 — dark grey text on white opaque badge (no border)
%              applied consistently on ALL squares (dark and light)
% Margin     : 5 mm printer-safe on all outer edges
%
% Board geometry:
%   sq=4.9 cm, board=8*4.9=39.2 cm (no outer border strip)
%   margin=0.5 cm
%   tile        = 20.5 x 29.2 cm
%   col_step    = 19.2 cm  =>  col_offsets = [0, 19.2]
%   row_step    = 10.5 cm  =>  row_offsets = [0, 10.5]
%   overlap_h   =  1.3 cm
%   overlap_v   = 18.7 cm
%
% Label positions (inside outermost squares):
%   File letters: rank 1 (bottom row, j=7) and rank 8 (top row, j=0, rot 180)
%   Rank numbers: file a (left col, i=0) and file h (right col, i=7, rot 180)
%   Corner squares carry both file and rank label
%
% Assembly: P1 top-left, P2 top-right, P3 bot-left, P4 bot-right
%   Align P1+P2 by matching shared squares in files d-e (1.3 cm overlap zone)
%   Align top+bottom rows by matching ranks 4-6 (18.7 cm overlap zone)

\documentclass[a4paper]{article}

\usepackage[
  a4paper,
  top=0mm, bottom=0mm, left=0mm, right=0mm,
  noheadfoot, nomarginpar
]{geometry}

\usepackage{tikz}
\usepackage{helvet}
\renewcommand{\familydefault}{\sfdefault}

\pagestyle{empty}

\def\sq{4.9}      % square size cm
\def\btotal{39.2} % total board size cm
\def\lpad{0.22}   % label inset from square edge cm

% -----------------------------------------------------------------------
% Crosshatch pattern macro (local coords 0,0 to sq,sq)
% Angles 45+135, spacing 3.0mm, step along axis = 3.0*sqrt(2) = 0.4243 cm
% -----------------------------------------------------------------------
\newcommand{\crosshatch}{%
  \foreach \cc in {
    -4.95,-4.53,-4.10,-3.67,-3.25,-2.82,-2.40,-1.97,-1.55,
    -1.12,-0.69,-0.27, 0.15, 0.58, 1.00, 1.43, 1.85, 2.28,
     2.70, 3.13, 3.55, 3.98, 4.40, 4.83}{%
    \draw[black,line width=0.30pt](0,\cc)--(\sq,\cc+\sq);%
    \draw[black,line width=0.30pt](0,\cc+\sq)--(\sq,\cc);}%
}

% -----------------------------------------------------------------------
% Label style 7: dark grey text on white opaque badge, no border
% \lbl{anchor}{rotation}{text}  — used for all label nodes
% -----------------------------------------------------------------------
\newcommand{\lbl}[3]{%
  \node[anchor=#1, rotate=#2,
        font=\fontsize{9}{10}\selectfont\bfseries,
        text=black!80,
        fill=white,
        inner sep=1.2pt] at
}
% Usage: \lbl{south}{0}{a} at (x,y);

% -----------------------------------------------------------------------
% \drawpage{xoffset}{yoffset}
%   Full A4 tikzpicture. Board origin at (0.5-xoff, 0.5-yoff).
%   y grows downward. All 64 squares drawn; dark squares get crosshatch
%   via a shifted+clipped scope immediately after their white base fill.
% -----------------------------------------------------------------------
\newcommand{\drawpage}[2]{%
  \pgfmathsetmacro{\ssx}{0.5 - #1}%
  \pgfmathsetmacro{\ssy}{0.5 - #2}%
  \noindent%
  \begin{tikzpicture}[x=1cm, y=-1cm]

    \useasboundingbox (0,0) rectangle (21,29.7);
    \clip (0,0) rectangle (21,29.7);
    \fill[white] (0,0) rectangle (21,29.7);

    % ---- 64 squares: light=white fill, dark=white+crosshatch ----
    \foreach \i in {0,...,7}{
      \foreach \j in {0,...,7}{
        \pgfmathsetmacro{\dk}{int(mod(\i+\j,2))}%
        \ifnum\dk=0
          % Light square — plain white
          \fill[white]
            ({\i*\sq+\ssx},{\j*\sq+\ssy})
            rectangle
            ({\i*\sq+\sq+\ssx},{\j*\sq+\sq+\ssy});
        \else
          % Dark square — white base then clipped crosshatch in local scope
          \fill[white]
            ({\i*\sq+\ssx},{\j*\sq+\ssy})
            rectangle
            ({\i*\sq+\sq+\ssx},{\j*\sq+\sq+\ssy});
          \begin{scope}[shift={({\i*\sq+\ssx},{\j*\sq+\ssy})}]
            \begin{scope}
              \clip (0,0) rectangle (\sq,\sq);
              \crosshatch
            \end{scope}
          \end{scope}
        \fi
      }
    }

    % ---- Grid lines ----
    \foreach \k in {0,...,8}{
      \draw[black, line width=0.3pt]
        ({\k*\sq+\ssx},{\ssy}) -- ({\k*\sq+\ssx},{8*\sq+\ssy});
      \draw[black, line width=0.3pt]
        ({\ssx},{\k*\sq+\ssy}) -- ({8*\sq+\ssx},{\k*\sq+\ssy});
    }

    % ---- Outer playing area border ----
    \draw[black, line width=1.2pt]
      ({\ssx},{\ssy}) rectangle ({8*\sq+\ssx},{8*\sq+\ssy});

    % ---- Coordinate labels (style 7: white badge, black!80 text) ----
    % All labels use the same style — badge is visible on both dark and light.

    % File labels — rank 1 row (j=7), normal orientation
    \foreach \i/\ltr in {0/a,1/b,2/c,3/d,4/e,5/f,6/g,7/h}{%
      \node[anchor=south,
            font=\fontsize{9}{10}\selectfont\bfseries,
            text=black!80, fill=white, inner sep=1.2pt]
        at ({\i*\sq+\sq/2+\ssx},{7*\sq+\sq-\lpad+\ssy}) {\ltr};%
    }

    % File labels — rank 8 row (j=0), rotated 180 for Black
    \foreach \i/\ltr in {0/a,1/b,2/c,3/d,4/e,5/f,6/g,7/h}{%
      \node[anchor=south, rotate=180,
            font=\fontsize{9}{10}\selectfont\bfseries,
            text=black!80, fill=white, inner sep=1.2pt]
        at ({\i*\sq+\sq/2+\ssx},{\lpad+\ssy}) {\ltr};%
    }

    % Rank labels — file a col (i=0), normal orientation
    \foreach \j/\num in {0/8,1/7,2/6,3/5,4/4,5/3,6/2,7/1}{%
      \node[anchor=west,
            font=\fontsize{9}{10}\selectfont\bfseries,
            text=black!80, fill=white, inner sep=1.2pt]
        at ({\lpad+\ssx},{\j*\sq+\sq/2+\ssy}) {\num};%
    }

    % Rank labels — file h col (i=7), rotated 180 for Black
    \foreach \j/\num in {0/8,1/7,2/6,3/5,4/4,5/3,6/2,7/1}{%
      \node[anchor=west, rotate=180,
            font=\fontsize{9}{10}\selectfont\bfseries,
            text=black!80, fill=white, inner sep=1.2pt]
        at ({7*\sq+\sq-\lpad+\ssx},{\j*\sq+\sq/2+\ssy}) {\num};%
    }

  \end{tikzpicture}%
}

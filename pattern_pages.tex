% Pattern sampler — 12 pages, one pattern per page
% Shows squares g1, g2, h1, h2 (49mm each) with labels and pattern fills.
%
% KEY FIX: each dark square uses a shifted scope so pattern coords (0,0)-(sq,sq)
% map exactly onto that square in global coords. Clip is also (0,0)-(sq,sq)
% within the shifted scope.
%
% Square layout (y grows downward):
%   top-left  = g2 (LIGHT) at (bx,      by     )
%   top-right = h2 (DARK)  at (bx+sq,   by     )
%   bot-left  = g1 (DARK)  at (bx,      by+sq  )
%   bot-right = h1 (LIGHT) at (bx+sq,   by+sq  )
%
% Labels (only on outermost squares of full board):
%   g1 (DARK):  file 'g'  bottom-centre, white
%   g2 (LIGHT): no label
%   h1 (LIGHT): file 'h'  bottom-centre, dark grey | rank '1' right-centre, dark grey, rot180
%   h2 (DARK):  rank '2'  right-centre, white, rot180

\def\sq{4.9}    % square size cm
\def\lpad{0.22} % label inset from edge cm

% -----------------------------------------------------------------------
% \patsquare{x0}{y0}{patcode}
%   Draws a dark square at global (x0,y0)-(x0+sq,y0+sq).
%   Opens a shifted+clipped scope so {patcode} uses local coords (0,0)-(sq,sq).
%   White base fill is included automatically.
% -----------------------------------------------------------------------
\newcommand{\patsquare}[3]{%
  \begin{scope}[shift={(#1,#2)}]
    % White base
    \fill[white] (0,0) rectangle (\sq,\sq);
    % Clipped pattern in local coords
    \begin{scope}
      \clip (0,0) rectangle (\sq,\sq);
      #3
    \end{scope}
  \end{scope}
}

% -----------------------------------------------------------------------
% \boardpage{sampnum}{title}{detail}{patcode}
% -----------------------------------------------------------------------
\newcommand{\boardpage}[4]{%
  \noindent
  \begin{tikzpicture}[x=1cm, y=-1cm]

    \useasboundingbox (0,0) rectangle (17.1, 23.7);

    % -- Title --
    \node[anchor=north west,
          font=\fontsize{16}{20}\selectfont\bfseries, text=black]
      at (0,0) {Sample #1 — #2};
    \node[anchor=north west,
          font=\fontsize{10}{13}\selectfont, text=black!55]
      at (0,1.05) {#3};

    % -- Board: centred horizontally, top at by=2.8cm --
    % hcentre = (17.1 - 2*sq)/2 = (17.1-9.8)/2 = 3.65
    \pgfmathsetmacro{\bx}{(17.1 - 2*\sq)/2}
    \def\by{2.8}

    % Light squares (plain white)
    \fill[white] (\bx,        \by)      rectangle (\bx+\sq,   \by+\sq);
    \fill[white] (\bx+\sq,    \by+\sq)  rectangle (\bx+2*\sq, \by+2*\sq);

    % Dark squares with pattern (shifted local scope)
    \patsquare{\bx+\sq}{\by}{#4}      % h2: top-right
    \patsquare{\bx}{\by+\sq}{#4}      % g1: bot-left

    % Grid lines
    \foreach \k in {0,1,2}{
      \draw[black, line width=0.4pt]
        (\bx+\k*\sq,\by) -- (\bx+\k*\sq,\by+2*\sq);
      \draw[black, line width=0.4pt]
        (\bx,\by+\k*\sq) -- (\bx+2*\sq,\by+\k*\sq);
    }
    \draw[black, line width=1.4pt]
      (\bx,\by) rectangle (\bx+2*\sq,\by+2*\sq);

    % Labels
    % g1 (bot-left, DARK): file 'g', bottom of square, white
    \node[anchor=south, font=\fontsize{12}{13}\selectfont\bfseries, text=white]
      at (\bx+\sq/2, \by+2*\sq-\lpad) {g};
    % h1 (bot-right, LIGHT): file 'h', bottom of square, dark
    \node[anchor=south, font=\fontsize{12}{13}\selectfont\bfseries, text=black!70]
      at (\bx+3*\sq/2, \by+2*\sq-\lpad) {h};
    % h1 (bot-right, LIGHT): rank '1', right of square, dark, rot180
    \node[anchor=west, rotate=180, font=\fontsize{12}{13}\selectfont\bfseries, text=black!70]
      at (\bx+2*\sq-\lpad, \by+3*\sq/2) {1};
    % h2 (top-right, DARK): rank '2', right of square, white, rot180
    \node[anchor=west, rotate=180, font=\fontsize{12}{13}\selectfont\bfseries, text=white]
      at (\bx+2*\sq-\lpad, \by+\sq/2) {2};

    % Description box
    \draw[black!20, line width=0.5pt, rounded corners=2pt]
      (0,\by+2*\sq+0.55) rectangle (17.1,\by+2*\sq+2.6);
    \fill[black!4, rounded corners=2pt]
      (0,\by+2*\sq+0.55) rectangle (17.1,\by+2*\sq+2.6);
    \node[anchor=north west, font=\fontsize{8}{10}\selectfont\bfseries, text=black!40]
      at (0.3,\by+2*\sq+0.85) {PATTERN SPECIFICATION};
    \node[anchor=north west, font=\fontsize{10.5}{14}\selectfont, text=black]
      at (0.3,\by+2*\sq+1.3) {#3};
    \node[anchor=north west, font=\fontsize{8.5}{11}\selectfont, text=black!50]
      at (0.3,\by+2*\sq+2.0)
      {Dark squares: g1 (file g, rank 1) and h2 (file h, rank 2)
       \enspace\textbullet\enspace Light squares: g2 and h1 (white, no ink)};

  \end{tikzpicture}
  \newpage
}
